CREATE DATABASE CINE
GO

USE CINE
GO

SET DATEFORMAT dmy

CREATE TABLE Tipos_Salas (
	id_tipo_sala int IDENTITY(1,1),
	descripcion nvarchar (50) NOT NULL,
	precio decimal(8,2)
	CONSTRAINT pk_tipo_sala PRIMARY KEY(id_tipo_sala)
)

CREATE TABLE Salas (
	id_sala int IDENTITY(1,1),
	descripcion nvarchar (50) NOT NULL,
	id_tipo_sala int,
	capacidad_sala int
	CONSTRAINT pk_salas PRIMARY KEY(id_sala)
	CONSTRAINT fk_sala_tiposala FOREIGN KEY(id_tipo_sala) REFERENCES Tipos_Salas(id_tipo_sala)
)

CREATE TABLE Clasificaciones (
	id_clasificacion int IDENTITY(1,1),
	descripcion nvarchar (50) NOT NULL
	CONSTRAINT pk_clasif PRIMARY KEY(id_clasificacion)
)

CREATE TABLE Paises (
	id_pais int IDENTITY(1,1),
	descripcion nvarchar(50)
	CONSTRAINT pk_paises PRIMARY KEY(id_pais)
)

CREATE TABLE Clientes (
	id_cliente int IDENTITY(1,1),
	nombre nvarchar(50) NOT NULL,
	apellido nvarchar(50) NOT NULL,
	dni int NOT NULL,
	fecha_nac date,
	calle_direccion nvarchar(50),
	nro_direccion int,
	nro_telefono int,
	email nvarchar(50), 
	nacionalidad int,
	fecha_alta datetime
	CONSTRAINT pk_clientes PRIMARY KEY(id_cliente),
	CONSTRAINT fk_cliente_paises FOREIGN KEY(nacionalidad) REFERENCES Paises(id_pais)
)

CREATE TABLE Generos_Peliculas (
	id_genero int IDENTITY(1,1),
	descripcion nvarchar(50)
	CONSTRAINT pk_genero PRIMARY KEY(id_genero)
)

CREATE TABLE Peliculas (
	id_pelicula int IDENTITY(1,1),
	nombre_pelicula nvarchar (50) NOT NULL,
	id_clasificacion int,
	id_genero int,
	doblaje bit,
	pais_origen int
	CONSTRAINT pk_peliculas PRIMARY KEY (id_pelicula),
	CONSTRAINT fk_peli_clasif FOREIGN KEY(id_clasificacion) REFERENCES Clasificaciones (id_clasificacion),
	CONSTRAINT fk_peli_paises FOREIGN KEY(pais_origen) REFERENCES Paises (id_pais),
	CONSTRAINT fk_pelicula_genero FOREIGN KEY(id_genero) REFERENCES Generos_Peliculas (id_genero)
)

CREATE TABLE Formas_Pago (
	id_forma_pago int IDENTITY(1,1),
	descripcion nvarchar(50) NOT NULL
	CONSTRAINT pk_forma_pago PRIMARY KEY(id_forma_pago)
)

CREATE TABLE Butacas (
	id_butaca int IDENTITY(1,1),
	fila char NOT NULL,
	numero int NOT NULL
	CONSTRAINT pk_butacas PRIMARY KEY(id_butaca)
)

CREATE TABLE Funciones (
	id_funcion int IDENTITY(1,1),
	id_sala int,
	id_pelicula int,
	fecha date NOT NULL,
	hora time NOT NULL
	CONSTRAINT pk_funciones PRIMARY KEY(id_funcion),
	CONSTRAINT fk_funciones_salas FOREIGN KEY(id_sala) REFERENCES Salas(id_sala),
	CONSTRAINT fk_funciones_peliculas FOREIGN KEY(id_pelicula) REFERENCES Peliculas(id_pelicula)
)

CREATE TABLE Tipos_Compras(
	id_tipo_compra int IDENTITY(1,1),
	descripcion nvarchar(50)
	CONSTRAINT pk_tipo_compra PRIMARY KEY(id_tipo_compra) 
)

CREATE TABLE Comprobantes (
	id_comprobante int IDENTITY(1,1),
	id_cliente int,
	id_tipo_compra int,
	fecha datetime NOT NULL,
	monto_total decimal (10,2)
	CONSTRAINT pk_comprob PRIMARY KEY(id_comprobante),
	CONSTRAINT fk_comp_cliente FOREIGN KEY(id_cliente) REFERENCES Clientes(id_cliente),
	CONSTRAINT fk_comp_tipo_comp FOREIGN KEY(id_tipo_compra) REFERENCES Tipos_Compras(id_tipo_compra)
)

CREATE TABLE Tipos_Entradas(
	id_tipo_entrada int IDENTITY(1,1),
	descripcion nvarchar(50),
	precio decimal(8,2)
	CONSTRAINT pk_tipo_entrada PRIMARY KEY(id_tipo_entrada)
)

CREATE TABLE Promociones(
	id_promocion int IDENTITY(1,1),
	descripcion nvarchar(50),
	porcentaje decimal(3,2)
	CONSTRAINT pk_promociones PRIMARY KEY(id_promocion)
)

CREATE TABLE Detalle_Comprobantes (
	id_detalle_comprobante int IDENTITY(1,1),
	id_comprobante int,
	id_funcion int,
	id_tipo_entrada int,
	id_promocion int
	CONSTRAINT pk_detalle_comp PRIMARY KEY(id_detalle_comprobante),
	CONSTRAINT fk_detcomp_comp FOREIGN KEY(id_comprobante) REFERENCES Comprobantes(id_comprobante),
	CONSTRAINT fk_detcomp_funciones FOREIGN KEY(id_funcion) REFERENCES Funciones(id_funcion),
	CONSTRAINT fk_detcomp_tipoent FOREIGN KEY(id_tipo_entrada) REFERENCES Tipos_Entradas(id_tipo_entrada),
	CONSTRAINT fk_detcomp_promos FOREIGN KEY(id_promocion) REFERENCES Promociones(id_promocion)
)

CREATE TABLE Tickets (
	id_ticket int IDENTITY(1,1),
	id_cliente int,
	id_detalle_comprobante int,
	id_butaca int,	
	reserva bit
	CONSTRAINT pk_tickets PRIMARY KEY(id_ticket),
	CONSTRAINT fk_ticket_cliente FOREIGN KEY(id_cliente) REFERENCES Clientes(id_cliente),
	CONSTRAINT fk_tick_butacas FOREIGN KEY(id_butaca) REFERENCES Butacas(id_butaca),
	CONSTRAINT fk_tick_comprob FOREIGN KEY(id_detalle_comprobante) REFERENCES Detalle_Comprobantes(id_detalle_comprobante)
)

CREATE TABLE Comprobantes_x_Formas_Pago (
	id_comprobante int,
	id_forma_pago int,
	monto decimal (10,2)
	CONSTRAINT pk_compxfp PRIMARY KEY(id_comprobante, id_forma_pago)
	CONSTRAINT fk_compxfp_comprob FOREIGN KEY(id_comprobante) REFERENCES Comprobantes(id_comprobante),
	CONSTRAINT fk_compxfp_fp FOREIGN KEY(id_forma_pago) REFERENCES Formas_Pago(id_forma_pago)
)
